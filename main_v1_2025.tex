\title{
Three dimensional toric varieties from quiver represenations
}


\date{}


\begin{document}
\begin{abstract}
For a given dimension, there are only a finite number of toric varieties that emerge as moduli spaces of quiver representations with dimension vectors all equal to one. They are known as toric quiver varieties. In this work, we classify all the three-dimensional toric quiver varieties. Specifically, we demonstrate that there are five isomorphism classes among them, and that a single variation of GIT with 18 chambers is responsible for generating all of them. Additionally, we explore and describe the birational transformations between these varieties from both combinatorial and moduli theory perspectives.
\end{abstract}

\begin{definition}
    Given a quiver $Q$, the canonical weight $\delta_Q$ is defined as the image under $I_Q$ of placing a flow of 1 on all edges of $Q$, that is, $\delta_Q := I_Q(\mathbf{1})$ where $\mathbf{1} \in \R^{Q_1}$.
\end{definition}

\begin{proposition}
    $\Delta(Q(0), \delta_{Q(0)})$ is the $d$-simplex.
\end{proposition}

\begin{proof}
    Note all spanning trees of $Q(0)$ are of the form $T_i := (Q(0)_0, \{\alpha_i\})$ for $1 \leq i \leq d + 1$ and that $\delta_{Q(0)} = (-(d + 1), d + 1)$. For any $T_i$,  the unique flow supported by the tree is $\varepsilon^{T_i} := (d + 1)e_i \in \R^{d+1}$. Then,
    \[
    \Delta(Q(0), \delta_{Q(0)}) = \conv\{\varepsilon^{T_i} \:|\: i \leq 1 \leq d + 1\}
    \]
    \noindent which is the $d$-simplex.
\end{proof}

\begin{proposition}
    For $1 \leq m \leq d + 1$, there exist $d + 1$ $w_0$-semistable spanning trees of $Q(m)$. Moreover, the flows supported on them are all equal.
\end{proposition}

\begin{lemma}
   Let $v_\varepsilon^\perp \in \R^{Q_0}$ be perpendicular to the wall $\:W$ from Proposition \ref{wall-existence}. For small enough $\varepsilon$, 
    \[
    \Delta(Q(m), \delta_{Q(m)}) \cong \Delta(Q(m+1), w_0 - v_\varepsilon^\perp)
    \]
\end{lemma}

\begin{proposition}
    There are $d$ $(w_0-v_\varepsilon^\perp)$-unstable trees and one $(w_0-v_\varepsilon^\perp)$-stable tree of $Q(m + 1)$.
\end{proposition}

\begin{proposition}
    For small enough $\varepsilon$, there exist $d$ $(w_0 + v_\varepsilon^\perp)$-stable trees of $Q(m+1)$ that are also $w_0$-semistable.
\end{proposition}

\begin{lemma}
    Let $T^1, T^2, \ldots, T^n$ be the $n$ $(w_0 + v_\varepsilon^\perp)$-stable trees of $Q(m+1)$ and $T^0$ the $(w_0 - v_\varepsilon^\perp)$-stable tree of $Q(m+1)$  that is also $w_0$-semistable. Then, for some $u \in \R^{d+2}$,
    \[
        \Delta(\widetilde{Q}, w_0) \cong \conv\{T^i \:|\: 0 \leq i \leq n\} + u
    \]
\end{lemma}

\begin{definition}

\begin{proposition}
    Denote the $d$ $(w_0 + v_\varepsilon^\perp)$-stable trees of $\widetilde{Q}$ that are also $w_0$-semistable as $S^1, S^2, \ldots, S^d$. Then, up to a coordinate change,
    \[
        \conv\{S^i \:|\: 1 \leq i \leq d\} = \Delta_{d-1}'
    \]
\end{proposition}

\begin{proposition}
    There exists a $w_0$ and $v_\varepsilon^\perp$ such that $\delta_{\widetilde{Q}} = w_0 + v_\varepsilon^\perp$.
\end{proposition}

\begin{definition}
    Let $R$ be a quiver. Let $Q'$ be a quiver constructed  by placing a sink in at most $d+1$ edges of $\hat{Q}$. If there exists a $Q'$ such that $\Delta(R, \delta_R) \cong \Delta(Q', \delta_{Q'})$, then we say that $\Gamma(R)$ is the number of sinks placed in $\hat{Q}$ to construct $Q'$.
\end{definition}

\begin{theorem}
    $\Delta(\widetilde{Q}, \delta_{\widetilde{Q}})$ is affinely isomorphic to the $d$-simplex with $\Gamma(\widetilde{Q})$ vertices cut off.
\end{theorem}

Here, we note that the ``cutting off'' is special. If $\Gamma(Q) = 1$, then cutting off a vertex is as expected. For subsequent cuts, the section formed by the cut contains as many newly created vertices from previous cuts as possible while still maintaining the $(d-1)$-simplex shape.
